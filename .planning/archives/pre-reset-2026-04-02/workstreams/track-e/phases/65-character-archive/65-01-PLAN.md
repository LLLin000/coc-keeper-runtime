    ```python
        profile = COCInvestigatorProfile(
            occupation="记者",
            age=34,
            san=65,
            hp=12,
            mp=8,
            luck=45,
            build=1,
            damage_bonus="1d4",

        )

            source=CharacterSourceInfo(provider="coc_pereg", label=CharacterSourceLabel.COC_PREGEN),
            external_id="inv-001",
            name="张记者",
            species="人类",
            coc=profile,
        )

        )

        assert record.source.label == CharacterSourceLabel.COC_PREGEN

        store = PersistenceStore(Path(":memory:"))
        profiles = {

        assert "inv-001" in loaded
    ```

    ```python
    """CHAR-02: Character profile projection into campaign session."""
    import pytest
    from dm_bot.orchestrator.session_store import SessionStore, SessionPhase

    @pytest.fixture
    def session_with_profiles():
        store = SessionStore()
        store.bind_campaign(campaign_id="c1", channel_id="ch1", guild_id="g1", owner_id="owner")
        store.join_campaign(channel_id="ch1", user_id="player1")
        store.join_campaign(channel_id="ch1", user_id="player2")
        session = store.get_by_channel("ch1")
        session.transition_to(SessionPhase.AWAITING_READY)
        return store

    def test_select_archive_profile_sets_active_character(session_with_profiles):
        store = session_with_profiles
        profiles = {
            "owner": {"name": "张记者", "coc": {"san": 65}},
            "player1": {"name": "王侦探", "coc": {"san": 50}},
        }
        for uid, data in profiles.items():
            store.select_archive_profile(channel_id="ch1", user_id=uid, profile_data=data)
        session = store.get_by_channel("ch1")
        assert session.active_characters.get("owner") == "张记者"
        assert session.active_characters.get("player1") == "王侦探"

    def test_multi_player_different_profiles(session_with_profiles):
        store = session_with_profiles
        store.select_archive_profile(channel_id="ch1", user_id="owner", profile_data={"name": "张记者"})
        store.select_archive_profile(channel_id="ch1", user_id="player1", profile_data={"name": "王侦探"})
        session = store.get_by_channel("ch1")
        assert len(session.active_characters) == 2
        assert session.active_characters["owner"] != session.active_characters["player1"]

    def test_profile_projection_survives_phase_transition(session_with_profiles):
        store = session_with_profiles
        store.select_archive_profile(channel_id="ch1", user_id="owner", profile_data={"name": "张记者"})
        session = store.get_by_channel("ch1")
        session.transition_to(SessionPhase.SCENE_ROUND_OPEN)
        assert session.active_characters.get("owner") == "张记者"
    ```
