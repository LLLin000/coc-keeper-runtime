<objective>
Fix PersistenceStore so that `:memory:` SQLite databases maintain their tables across all operations, eliminating "no such table: campaign_sessions" and "no such table: campaign_state" errors in scenario runs.

Purpose: The root cause is that `PersistenceStore._connect()` opens a NEW connection for every operation. For `:memory:` SQLite, each new connection is a separate database — `_init_db()` creates tables in one connection, but `save_campaign_state()` opens a different connection where tables don't exist.
Output: Fixed connection management, updated test driver, passing tests.
</objective>

```python
class PersistenceStore:
    def __init__(self, path: str | Path) -> None:
        # path can be ":memory:" for in-memory, or a file path
        self._path = Path(path)
        self._init_db()

    def _connect(self) -> sqlite3.Connection:
        # BUG: Opens new connection each call — for :memory: this creates separate databases
        path_str = str(self._path)
        if path_str == ":memory:":
            return sqlite3.connect("file::memory:?cache=shared", uri=True)
        return sqlite3.connect(path_str)

    def _init_db(self) -> None:
        # Creates tables: campaign_state, campaign_sessions, campaign_events, archive_profiles
        with self._connect() as conn:
            conn.execute("create table if not exists campaign_state ...")
            conn.execute("create table if not exists campaign_sessions ...")
            conn.execute("create table if not exists campaign_events ...")
            conn.execute("create table if not exists archive_profiles ...")

    # All CRUD methods follow same pattern: open new connection, execute, close
    def save_campaign_state(self, campaign_id: str, state: dict) -> None:
        with self._connect() as conn: ...
    def load_campaign_state(self, campaign_id: str) -> dict:
        with self._connect() as conn: ...
    def save_sessions(self, sessions: dict) -> None:
        with self._connect() as conn: ...
    def load_sessions(self) -> dict:
        with self._connect() as conn: ...
```

    Modify `_connect()` to:
    1. If `self._conn` is not None (memory mode), return `self._conn`
    2. Otherwise, use existing logic: `sqlite3.connect(path_str)`

    ```python
    def close(self) -> None:
        if self._conn is not None:
            self._conn.close()
            self._conn = None
    ```

    Update `_init_db()` to use `self._connect()` (which now returns the shared connection for memory mode).

    CRITICAL: Do NOT change table schemas, column names, or any SQL statements. Only change connection lifecycle.
