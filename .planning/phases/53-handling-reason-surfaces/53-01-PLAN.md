def test_bind_duplicate_character_instance_rejected():
    """A user cannot create a second active character instance in the same campaign."""
    store = SessionStore()
    store.bind_campaign(campaign_id="camp-1", channel_id="chan-1", guild_id="guild-1", owner_id="user-1")
    store.join_campaign(channel_id="chan-1", user_id="user-2")
    
    # First character bind succeeds
    store.bind_character(channel_id="chan-1", user_id="user-2", character_name="Alice")
    
    # Second bind should raise ValidationError
    with pytest.raises(ValidationError) as exc_info:
        store.bind_character(channel_id="chan-1", user_id="user-2", character_name="Bob")
    
    assert "already has an active character" in str(exc_info.value).lower()


def test_bind_duplicate_instance_allowed_with_override():
    """A user CAN replace existing instance if they provide explicit override."""
    store = SessionStore()
    store.bind_campaign(campaign_id="camp-1", channel_id="chan-1", guild_id="guild-1", owner_id="user-1")
    store.join_campaign(channel_id="chan-1", user_id="user-2")
    
    store.bind_character(channel_id="chan-1", user_id="user-2", character_name="Alice")
    
    # Should succeed with force=True override
    store.bind_character(channel_id="chan-1", user_id="user-2", character_name="Bob", replace_existing=True)
    
    # Verify new character is active
    session = store.get_by_channel("chan-1")
    assert session.members["user-2"].active_character_name == "Bob"
```

3. Update `SessionStore.bind_character()` method:

   def bind_character(
       self,
       channel_id: str,
       user_id: str,
       character_name: str,
       replace_existing: bool = False,
   ) -> None:
       session = self._get_session_or_raise(channel_id)
       
       # Guard: Check user is a member
       if user_id not in session.members:
           raise ValidationError(f"User {user_id} is not a member. Join the campaign first.")
       
       # Guard: Check no duplicate instance (unless replace_existing)
       if not replace_existing and user_id in session.character_instances:
           existing = session.character_instances[user_id]
           raise ValidationError(
               f"User {user_id} already has active character '{existing.character_name}'. "
               f"Use /ready with replace_existing=True to replace, or use a different character name."
           )
       
       # ... rest of existing logic
       instance = CampaignCharacterInstance(
           campaign_id=session.campaign_id,
           user_id=user_id,
           character_name=character_name,
       )
       session.character_instances[user_id] = instance
       session.members[user_id].active_character_name = character_name
   ```

4. Run `uv run pytest tests/test_membership_gates.py -x` — expect ALL PASS (GREEN)

**Verify:** `uv run pytest tests/test_membership_gates.py -x`
